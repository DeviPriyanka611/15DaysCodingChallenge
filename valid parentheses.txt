class Solution {
    public boolean isValid(String s) {
        Stack<Character> p = new Stack<>();
        
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            
            if (ch == '(' || ch == '{' || ch == '[') {
                p.push(ch);
               } 
               else {
                if (p.isEmpty()) return false;

                if (ch == ')' && p.peek() == '(')
                 {
                    p.pop();
                }
                 else if (ch == '}' && p.peek() == '{') {
                    p.pop();
                }
                 else if (ch == ']' && p.peek() == '[') {
                    p.pop();
                } 
                else {
                    return false;
                }
            }
        }
        
        return p.isEmpty();
    }

    public static void main(String args[]) {
        Solution obj = new Solution();
        System.out.println(obj.isValid("()"));
    }
}
